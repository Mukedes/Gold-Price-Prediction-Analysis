# Rolling Mean
df['Rolling_Mean'] = df['Close'].rolling(window=30).mean()

plt.figure(figsize=(12, 6))
plt.plot(df['Close'], label='Close Price')
plt.plot(df['Rolling_Mean'], label='30-Day Rolling Mean', color='orange')
plt.title('Gold Prices with 30-Day Rolling Mean')
plt.xlabel('Date')
plt.ylabel('Close Price')
plt.legend()
plt.show()

# Decompose the time series
result = seasonal_decompose(df['Close'], model='multiplicative', period=365)
result.plot()
plt.show()
